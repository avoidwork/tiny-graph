/**
 * Tiny graph data structure for Client or Server
 *
 * @author Jason Mulligan <jason.mulligan@avoidwork.com>

 * @license BSD-3 <https://raw.github.com/avoidwork/tiny-graph/master/LICENSE>
 * @link http://avoidwork.github.io/tiny-graph
 * @module tiny-graph
 * @version 1.0.0
 */
( function ( global ) {
"use strict";

/**
 * Node
 *
 * @param {Mixed} value [Optional] Node value
 * @constructor
 */
function Node (value) {
	this.value = value || null;
	this.edges = {};
}

/**
 * Setting constructor loop
 *
 * @type {Node}
 */
Node.prototype.constructor = Node;

/**
 * Node factory
 *
 * @method node
 * @param {Mixed} value [Optional] Node value
 * @return {Object} Node
 */
function node (value) {
	return new Node(value);
}

function Graph () {
	this.nodes = {}
}

Graph.prototype.constructor = Graph;

Graph.prototype.adjacent = function (x, y) {
	var n = this.nodes[x];

	if (n) {
		return n.edges[y] !== undefined;
	}

	return false;
};

Graph.prototype.neighbors = function (x) {
	var n = this.nodes[x];

	if (n) {
		return Object.keys(n.edges);
	}

	return [];
};

Graph.prototype.add = function (x, y) {
	var n = this.nodes[x];

	if ( n ) {
		n.edges[y] = null;
		return this;
	}

	return null;
};

Graph.prototype.delete = function (x, y) {
	var n = this.nodes[x];

	if ( n ) {
		delete n.edges[y];
	}

	return this;
}

Graph.prototype.get_node_value = function (x) {
	var n = this.nodes[x];

	if ( n ) {
		return n.value;
	}

	return null;
};

Graph.prototype.set_node_value = function (x, v) {
	var n = this.nodes[x];

	if ( n ) {
		n.value = v;
	}
	else {
		this.nodes[x] = node(v);
	}

	return this;
};

Graph.prototype.get_edge_value = function (x, y) {
	var n = this.nodes[x];

	if ( n && n.edges[y] ) {
		return n.edges[y];
	}

	return null;
};

Graph.prototype.set_edge_value = function (x, y, v) {
	var n = this.nodes[x];

	if ( n && n.edges[y] ) {
		n.edges[y] = v;
		return this;
	}

	return null;
}

function graph () {
	return new Graph();
}

// Node, AMD & window supported
if ( typeof exports != "undefined" ) {
	module.exports = graph;
}
else if ( typeof define == "function" ) {
	define( function () {
		return graph;
	} );
}
else {
	global.graph = graph;
}
} )( this );
